-- Create movie table
CREATE TABLE movie (
  mID NUMBER PRIMARY KEY,
  Title VARCHAR2(100),
  Year NUMBER,
  Director VARCHAR2(50)
);

-- Create users table (avoid 'user' since it's a reserved word)
CREATE TABLE users (
  uID NUMBER PRIMARY KEY,
  Name VARCHAR2(50)
);

-- Create rating table without auto_increment, use sequence for ratingID
CREATE TABLE rating (
  ratingID NUMBER PRIMARY KEY,
  uID NUMBER,
  mID NUMBER,
  Rating NUMBER CHECK (Rating BETWEEN 1 AND 5),
  ratingDate DATE,
  FOREIGN KEY (uID) REFERENCES users(uID),
  FOREIGN KEY (mID) REFERENCES movie(mID)
);

-- Sequence for ratingID
CREATE SEQUENCE rating_seq START WITH 1 INCREMENT BY 1;

-- Sample Queries
-- 1. Movies released after 2000
SELECT Title, Year
FROM movie
WHERE Year > 2000;

-- 2. Movies before 2000 with rating > 2
SELECT m.Title, m.mID, r.Rating
FROM movie m
JOIN rating r ON m.mID = r.mID
WHERE m.Year < 2000
AND r.Rating > 2;

-- 3. All ratings with user and movie info, ordered by rating desc
SELECT r.ratingID,
       u.Name AS UserName,
       m.Title AS MovieTitle,
       r.Rating,
       r.ratingDate
FROM rating r
JOIN users u ON r.uID = u.uID
JOIN movie m ON r.mID = m.mID
ORDER BY r.Rating DESC;

-- 4. Ratings grouped with concatenated movie titles using LISTAGG
SELECT r.Rating,
       LISTAGG(m.Title, ', ') WITHIN GROUP (ORDER BY m.Title) AS Movies
FROM rating r
JOIN movie m ON r.mID = m.mID
GROUP BY r.Rating
HAVING COUNT(DISTINCT m.mID) > 1;

-- 5. Distinct movies with rating > 4
SELECT DISTINCT m.mID, m.Title, m.Director
FROM movie m
JOIN rating r ON m.mID = r.mID
WHERE r.Rating > 4;
